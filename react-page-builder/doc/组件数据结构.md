# 组件数据结构

```json
{
    //是否是浏览器原生的组件
    isNative: true | false,
    //原生组件的tagName，或者自定义组件
    type: 'div' | 'img' | 'Slider' | 'RegisterBox',
    //组件的中文名字，方便识别
    name: '组件名',
    //是否接受子组件：true表示可以接纳任何组件；数组表示，支持数组内的type
    acceptChild: boolean | Array<type>,
    //当前实例的子组件ID数组
    childs: [],
    //当前实例的ID
    cid: '组件惟一ID',
    //当前实例父组件的ID
    parentCid: '父组件惟一ID',
    //当前实例的style
    style: {

    },
    //组件支持设置哪些style
    styleDef: {

    },
    //当前实例的数据
    data: {

    },
    //当前实例支持设置哪些数据
    dataDef: {

    },
}
```
